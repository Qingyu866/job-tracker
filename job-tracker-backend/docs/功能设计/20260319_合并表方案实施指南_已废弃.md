```java
/**
 * 确保技能标签存在于数据库中
 * <p>
 * 如果技能不存在，使用 AI 生成技能信息并批量插入
 * </p>
 *
 * @param skillNames 技能名称列表
 * @return 技能名称 -> 技能ID 的映射
 */
private Map<String, Long> ensureSkillsExist(List<String> skillNames) {
    Map<String, Long> skillMap = new HashMap<>();

    // 1. 查询已存在的技能
    List<SkillTag> existingSkills = skillTagMapper.selectList(
        new LambdaQueryWrapper<SkillTag>()
            .in(SkillTag::getSkillName, skillNames)
    );

    // 2. 构建已存在技能的映射
    for (SkillTag skill : existingSkills) {
        skillMap.put(skill.getSkillName(), skill.getSkillId());
    }

    // 3. 找出缺失的技能
    List<String> missingSkills = skillNames.stream()
        .filter(name -> !skillMap.containsKey(name))
        .collect(Collectors.toList());

    // 4. 如果没有缺失的，直接返回
    if (missingSkills.isEmpty()) {
        return skillMap;
    }

    // 5. 使用 AI 生成缺失的技能信息
    log.info("检测到 {} 个缺失的技能，使用 AI 生成：{}", missingSkills.size(), missingSkills);

    List<SkillTag> newSkills = generateSkillTagsWithAI(missingSkills);

    // 6. 批量插入新技能
    for (SkillTag skill : newSkills) {
        skillTagMapper.insert(skill);
        skillMap.put(skill.getSkillName(), skill.getSkillId());
        log.info("创建新技能：{} (ID: {})", skill.getSkillName(), skill.getSkillId());
    }

    return skillMap;
}

/**
 * 使用 AI 生成技能标签
 *
 * @param skillNames 技能名称列表
 * @return 技能标签列表
 */
private List<SkillTag> generateSkillTagsWithAI(List<String> skillNames) {
    String prompt = buildSkillGenerationPrompt(skillNames);

    String aiResponse = chatLanguageModel.generate(prompt);
    String cleanedResponse = cleanMarkdownCodeBlocks(aiResponse);

    try {
        return objectMapper.readValue(
            cleanedResponse,
            new TypeReference<List<SkillTag>>() {}
        );
    } catch (Exception e) {
        log.error("AI 生成技能失败，使用默认值", e);
        // 返回只有名称的默认技能
        return skillNames.stream()
            .map(name -> SkillTag.builder()
                .skillName(name)
                .category("未分类")
                .description("AI 自动生成")
                .difficultyLevel(3)
                .build())
            .collect(Collectors.toList());
    }
}
```

```java
@Override
public MockInterviewSession createSession(CreateSessionRequest request) {
    // ... 现有代码 ...

    // ==================== 新增：生成考察计划 ====================

    // 1. 调用副面试官生成考察计划（25 轮）
    String context = buildContextForViceInterviewer(session);
    List<QuestionPlanDTO> plans = agents.viceInterviewer().generateQuestionPlan(context);

    log.info("副面试官生成了 {} 轮考察计划", plans.size());

    // 2. 确保所有技能存在于数据库中
    List<String> skillNames = plans.stream()
        .map(QuestionPlanDTO::getSkillName)
        .distinct()
        .collect(Collectors.toList());

    Map<String, Long> skillMap = ensureSkillsExist(skillNames);

    // 3. 批量插入考察计划（状态为 PENDING）
    for (QuestionPlanDTO plan : plans) {
        MockInterviewEvaluation evaluation = MockInterviewEvaluation.builder()
            .sessionId(session.getSessionId())
            .roundNumber(plan.getRoundNumber())
            .skillId(skillMap.get(plan.getSkillName()))
            .skillName(plan.getSkillName())
            .planStatus("PENDING")  // ⭐ 关键：状态为待考察
            .topicSource(plan.getTopicSource())
            .questionType(plan.getQuestionType())
            .plannedDifficulty(plan.getDifficulty())
            .contextInfo(plan.getContextInfo())
            .reason(plan.getReason())
            .build();

        evaluationMapper.insert(evaluation);
    }

    // 4. 更新会话进度信息
    session.setTotalPlans(plans.size());
    session.setCompletedPlans(0);
    session.setCurrentPlanId(
        plans.isEmpty() ? null :
        evaluationMapper.selectOne(
            new LambdaQueryWrapper<MockInterviewEvaluation>()
                .eq(MockInterviewEvaluation::getSessionId, session.getSessionId())
                .eq(MockInterviewEvaluation::getPlanStatus, "PENDING")
                .orderByAsc(MockInterviewEvaluation::getRoundNumber)
                .last("LIMIT 1")
        ).getEvaluationId()
    );

    sessionMapper.updateById(session);

    log.info("会话 {} 创建成功，共 {} 轮考察计划", session.getSessionId(), plans.size());

    return session;
}
```

```java
/**
 * 暂停面试
 *
 * @param sessionId 会话ID
 * @return 操作结果
 */
@Transactional
public Result<Void> pauseInterview(String sessionId) {
    MockInterviewSession session = sessionMapper.selectById(sessionId);

    if (session == null) {
        return Result.error("会话不存在");
    }

    if ("PAUSED".equals(session.getState())) {
        return Result.error("会话已暂停");
    }

    // 1. 更新会话状态
    session.setState("PAUSED");
    session.setPausedAt(LocalDateTime.now());
    sessionMapper.updateById(session);

    // 2. 更新当前执行中的计划状态
    evaluationMapper.update(null,
        new LambdaUpdateWrapper<MockInterviewEvaluation>()
            .eq(MockInterviewEvaluation::getSessionId, sessionId)
            .eq(MockInterviewEvaluation::getPlanStatus, "IN_PROGRESS")
            .set(MockInterviewEvaluation::getPlanStatus, "PAUSED")
    );

    // 3. 持久化所有 Agent 记忆
    agentFactory.persistMemories(sessionId);

    log.info("会话 {} 已暂停", sessionId);

    return Result.success("面试已暂停");
}
```

```java
/**
 * 恢复面试
 *
 * @param sessionId 会话ID
 * @return 操作结果
 */
@Transactional
public Result<Void> resumeInterview(String sessionId) {
    MockInterviewSession session = sessionMapper.selectById(sessionId);

    if (session == null) {
        return Result.error("会话不存在");
    }

    if (!"PAUSED".equals(session.getState())) {
        return Result.error("会话未暂停，无法恢复");
    }

    // 1. 恢复所有 Agent 记忆
    agentFactory.restoreMemories(sessionId);

    // 2. 恢复暂停中的计划状态
    evaluationMapper.update(null,
        new LambdaUpdateWrapper<MockInterviewEvaluation>()
            .eq(MockInterviewEvaluation::getSessionId, sessionId)
            .eq(MockInterviewEvaluation::getPlanStatus, "PAUSED")
            .set(MockInterviewEvaluation::getPlanStatus, "PENDING")
    );

    // 3. 更新会话状态
    session.setState("IN_PROGRESS");
    session.setResumedAt(LocalDateTime.now());
    sessionMapper.updateById(session);

    log.info("会话 {} 已恢复", sessionId);

    return Result.success("面试已恢复");
}
```

```java
@GetMapping("/sessions/{sessionId}/progress")
public Result<Map<String, Object>> getProgress(@PathVariable String sessionId) {

    // 查询所有计划

            planMap.put("skillName", plan.getSkillName());
            planMap.put("status", plan.getPlanStatus());

                planMap.put("totalScore", plan.getTotalScore());

        })

    progress.put("currentRound", session.getCurrentRound());
    progress.put("totalRounds", session.getTotalRounds());

    return Result.success(progress);
}
```

### 6.2 每轮对话时

```
POST /api/mock-interview/sessions/abc123/messages
{
  "content": "用户回答"
}

数据库：
1. 当前计划状态：PENDING → IN_PROGRESS → COMPLETED
2. 会话 completed_plans：0 → 1
3. 会话 current_plan_id：指向下一个计划
```

### 6.3 暂停/恢复时

```
POST /api/mock-interview/sessions/abc123/pause
→ 会话状态：IN_PROGRESS → PAUSED
→ 计划状态：IN_PROGRESS → PAUSED

POST /api/mock-interview/sessions/abc123/resume
→ 会话状态：PAUSED → IN_PROGRESS
→ 计划状态：PAUSED → PENDING
```
