## 📋 需求背景

### 需求描述
当前 AI 聊天系统仅支持纯文本输入。用户希望能够发送图片给 AI，让 AI 分析图片内容（如简历截图、职位描述图片、公司 logo 等），同时支持图片 + 文字的混合输入。

### 典型使用场景
1. **简历分析**：用户上传简历截图，AI 分析简历内容
2. **职位解析**：用户发送职位描述图片，AI 提取关键信息
3. **Logo 识别**：用户上传公司 logo，AI 识别公司信息
4. **混合输入**：用户发送图片 + 文字说明，AI 综合分析

### 业务价值
- 提升用户体验：更自然的交互方式
- 扩展 AI 能力：支持视觉理解
- 降低输入门槛：图片比文字更直观

## 🎯 技术方案

### 方案概述

采用**本地存储 + 按需转 Base64** 的生产级方案，优于 Base64 直传数据库：

| 维度 | Base64 存 DB | **本地存文件 + 路径存 DB** |
|------|-------------|---------------------|
| 数据库性能 | ❌ 差 | ✅ **优** |
| 模型兼容性 | ✅ 好 | ✅ **好** |
| 隐私安全 | ✅ 好 | ✅ **好** |
| 前端体验 | ✅ 快 | ⚠️ **中** |
| 扩展性 | ❌ 差 | ✅ **中** |

**核心思路**：
1. **前端**：选择图片 → HTTP 上传 → 获得 publicUrl → 预览显示
2. **后端存储**：接收文件 → 保存到本地磁盘 → 返回 filePath 和 publicUrl
3. **数据库**：存储 filePath 和 publicUrl（轻量级字符串）
4. **AI 调用**：读取本地文件 → 转换为 Base64 → 调用模型

#### 1.2 文件存储服务

**文件**：`src/main/java/com/jobtracker/service/FileStorageService.java`

```java
package com.jobtracker.service;

import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Service;
import org.springframework.web.multipart.MultipartFile;

import java.io.File;
import java.io.IOException;
import java.nio.file.Files;
import java.nio.file.Path;
import java.nio.file.Paths;
import java.time.LocalDate;
import java.time.format.DateTimeFormatter;
import java.util.UUID;

/**
 * 文件存储服务
 * <p>
 * 负责将上传的文件保存到本地磁盘
 * </p>
 */
@Slf4j
@Service
public class FileStorageService {

    @Value("${file.upload-dir:./uploads/chat}")
    private String baseUploadDir;

    @Value("${app.api-base-url:/api}")
    private String apiBaseUrl;

    /**
     * 保存上传的文件
     *
     * @param file 上传的文件
     * @return 文件访问信息
     */
    public ImageAttachment saveFile(MultipartFile file) throws IOException {
        // 1. 校验文件类型
        String contentType = file.getContentType();
        if (contentType == null || !contentType.startsWith("image/")) {
            throw new IllegalArgumentException("只支持图片文件");
        }

        // 2. 生成唯一文件名
        String originalFilename = file.getOriginalFilename();
        String extension = originalFilename != null && originalFilename.contains(".")
            ? originalFilename.substring(originalFilename.lastIndexOf("."))
            : ".jpg";
        String uniqueFileName = UUID.randomUUID().toString() + extension;

        // 3. 按日期组织目录：uploads/chat/2026-03-15/
        String dateDir = LocalDate.now().format(DateTimeFormatter.ofPattern("yyyy-MM-dd"));
        String relativeDir = "chat" + File.separator + dateDir + File.separator;
        String fullDir = baseUploadDir + File.separator + relativeDir;

        // 4. 确保目录存在
        File directory = new File(fullDir);
        if (!directory.exists()) {
            directory.mkdirs();
        }

        // 5. 保存文件
        File targetFile = new File(fullDir, uniqueFileName);
        file.transferTo(targetFile);

        log.info("文件已保存：path={}, size={}", targetFile.getAbsolutePath(), file.getSize());

        // 6. 构建返回信息
        String relativePath = relativeDir + uniqueFileName;  // 相对路径
        String publicUrl = apiBaseUrl + "/chat/images/" + relativeDir + uniqueFileName;

        return ImageAttachment.builder()
                .filePath(relativePath)
                .publicUrl(publicUrl)
                .mimeType(contentType)
                .fileName(originalFilename)
                .fileSize(file.getSize())
                .build();
    }

    /**
     * 读取文件并转换为 Base64
     *
     * @param filePath 相对路径
     * @return Base64 编码的数据
     */
    public byte[] readFileAsBytes(String filePath) throws IOException {
        String fullPath = baseUploadDir + File.separator + filePath;
        return Files.readAllBytes(Paths.get(fullPath));
    }

    /**
     * 获取文件的完整路径
     *
     * @param filePath 相对路径
     * @return 完整路径
     */
    public String getFullPath(String filePath) {
        return baseUploadDir + File.separator + filePath;
    }
}
```

```java
/**
 * 聊天控制器
 */
@Slf4j
@RestController
@RequestMapping("/api/chat")
@RequiredArgsConstructor
public class ChatController {

    private final FileStorageService fileStorageService;

    /**
     * 上传图片
     *
     * @param file 上传的文件
     * @return 图片访问信息
     */
    @PostMapping("/upload/image")
    public ResponseEntity<ImageAttachment> uploadImage(@RequestParam("file") MultipartFile file) {
        try {
            ImageAttachment attachment = fileStorageService.saveFile(file);
            return ResponseEntity.ok(attachment);
        } catch (IOException e) {
            log.error("文件上传失败", e);
            return ResponseEntity.internalServerError().build();
        }
    }

    /**
     * 获取图片（用于前端展示）
     *
     * @param relativePath 相对路径
     * @return 图片文件
     */
    @GetMapping("/images/**")
    public ResponseEntity<byte[]> getImage(@PathVariable String relativePath) {
        try {
            byte[] imageBytes = fileStorageService.readFileAsBytes(relativePath);

            // 从路径中提取文件扩展名来确定 MIME 类型
            String extension = relativePath.substring(relativePath.lastIndexOf(".") + 1);
            String mimeType = "image/" + extension;

            return ResponseEntity.ok()
                    .contentType(org.springframework.http.MediaType.parseMediaType(mimeType))
                    .body(imageBytes);
        } catch (IOException e) {
            log.error("读取图片失败：path={}", relativePath, e);
            return ResponseEntity.notFound().build();
        }
    }
}
```

#### 1.7 扩展 JobAgent 接口

**文件**：`src/main/java/com/jobtracker/agent/JobAgent.java`

```java
import dev.langchain4j.data.message.ImageContent;
import dev.langchain4j.data.message.UserMessage;
import dev.langchain4j.model.chat.ChatLanguageModel;
import dev.langchain4j.service.SystemMessage;
import dev.langchain4j.service.UserMessage;
import org.springframework.stereotype.Component;

import java.util.ArrayList;
import java.util.Base64;
import java.util.List;

public interface JobAgent {

    String chat(String userMessage, String currentDate, String currentTime, String dayOfWeek);

    /**
     * 多模态聊天（文字 + 图片）
     *
     * @param userMessage 用户文字消息
     * @param images 图片附件列表（包含路径信息）
     * @param currentDate 当前日期
     * @param currentTime 当前时间
     * @param dayOfWeek 星期几
     * @return AI 响应
     */
    default String chatWithImages(
            String userMessage,
            List<ImageAttachment> images,
            String currentDate,
            String currentTime,
            String dayOfWeek
    ) {
        if (images == null || images.isEmpty()) {
            return chat(userMessage, currentDate, currentTime, dayOfWeek);
        }

        // 构建多模态消息
        UserMessage.Builder messageBuilder = UserMessage.builder();

        // 添加文字内容
        if (userMessage != null && !userMessage.isEmpty()) {
            messageBuilder.text(userMessage);
        }

        // 添加图片内容
        for (ImageAttachment image : images) {
            try {
                // 读取文件并转换为 Base64（在实现类中注入 FileStorageService）
                byte[] imageBytes = readFileBytes(image.getFilePath());
                String base64 = Base64.getEncoder().encodeToString(imageBytes);

                messageBuilder.image(
                        ImageContent.fromBase64(base64, image.getMimeType())
                );
            } catch (Exception e) {
                // 记录错误但继续处理其他图片
                System.err.println("处理图片失败：" + image.getFilePath());
            }
        }

        // 构建并发送
        return chatWithUserMessage(messageBuilder.build(), currentDate, currentTime, dayOfWeek);
    }
}
```

```java
@Override
protected void handleTextMessage(WebSocketSession session, TextMessage message) throws Exception {
    String sessionId = session.getId();
    String payload = message.getPayload();

    try {
        WebSocketMessage wsMessage = objectMapper.readValue(payload, WebSocketMessage.class);
        String messageType = wsMessage.getType();
        String content = wsMessage.getContent();
        List<ImageAttachment> images = wsMessage.getImages();

        // 验证：内容和图片不能同时为空
        if ((content == null || content.trim().isEmpty()) &&
            (images == null || images.isEmpty())) {
            WebSocketMessage errorMessage = WebSocketMessage.error("消息内容和图片不能同时为空");
            session.sendMessage(new TextMessage(objectMapper.writeValueAsString(errorMessage)));
            return;
        }

        String chatSessionKey = wsMessage.getSessionKey() != null ? wsMessage.getSessionKey() : sessionId;

        // 1. 保存用户消息（包含图片路径）
        chatHistoryService.saveUserMessage(chatSessionKey, content, images);

        // 2. 调用 AI Agent
        JobAgent jobAgent = jobAgentFactory.createAgent(chatSessionKey);
        String aiResponse;

        if (images != null && !images.isEmpty()) {
            // 多模态调用
            aiResponse = jobAgent.chatWithImages(content, images, currentDate, currentTime, dayOfWeek);
        } else {
            // 纯文本调用
            aiResponse = jobAgent.chat(content, currentDate, currentTime, dayOfWeek);
        }

        // 3. 保存 AI 响应
        chatHistoryService.saveAssistantMessage(chatSessionKey, aiResponse);

        // 4. 发送响应
        WebSocketMessage responseMessage = WebSocketMessage.chat(aiResponse);
        session.sendMessage(new TextMessage(objectMapper.writeValueAsString(responseMessage)));

    } catch (Exception e) {
        log.error("处理消息失败：sessionId={}", sessionId, e);
        WebSocketMessage errorMessage = WebSocketMessage.error("处理失败：" + e.getMessage());
        session.sendMessage(new TextMessage(objectMapper.writeValueAsString(errorMessage)));
    }
}
```

#### 2.2 图片上传组件

**文件**：`frontend/src/components/chat/ImageUploader.tsx`

功能：
- 图片选择和预览
- **先 HTTP 上传**，获得 `imageId` 后再预览
- 支持删除已上传的图片
- 显示上传进度

**注意**：图片上传需要先创建消息，获取 `messageId` 后再上传。

```typescript
import { X, Image as ImageIcon, Loader2 } from 'lucide-react';
import { ImageAttachment } from '../../types/chat';

interface ImageUploaderProps {
  images: ImageAttachment[];
  onImagesChange: (images: ImageAttachment[]) => void;
  messageId?: number;       // 关联的消息ID（上传时需要）
  disabled?: boolean;
  maxImages?: number;
  maxSize?: number;  // MB
}

export const ImageUploader: React.FC<ImageUploaderProps> = ({
  images,
  onImagesChange,
  messageId,
  disabled = false,
  maxImages = 4,
  maxSize = 10
}) => {
  const fileInputRef = useRef<HTMLInputElement>(null);
  const [uploading, setUploading] = useState<Record<number, boolean>>({});

  const handleFileSelect = async (e: React.ChangeEvent<HTMLInputElement>) => {
    const files = Array.from(e.target.files || []);

    for (let i = 0; i < files.length && images.length + i < maxImages; i++) {
      const file = files[i];
      const index = images.length + i;

      // 验证文件类型
      if (!file.type.startsWith('image/')) {
        toast.error('请选择图片文件');
        continue;
      }

      // 验证文件大小
      if (file.size > maxSize * 1024 * 1024) {
        toast.error(`图片大小不能超过 ${maxSize}MB`);
        continue;
      }

      // 设置上传状态
      setUploading(prev => ({ ...prev, [index]: true }));

      try {
        // HTTP 上传图片（需要 messageId）
        if (!messageId) {
          throw new Error('消息ID不存在，请先发送消息');
        }

        const formData = new FormData();
        formData.append('file', file);
        formData.append('messageId', messageId.toString());

        const response = await chatApi.uploadImage(formData);

        // 后端返回：{ id, publicUrl, mimeType, fileName, fileSize }
        const attachment: ImageAttachment = {
          id: response.data.id,
          publicUrl: response.data.publicUrl,
          mimeType: response.data.mimeType,
          fileName: file.name,
          fileSize: file.size
        };

        // 添加到列表
        onImagesChange([...images, attachment]);

      } catch (error) {
        console.error('图片上传失败', error);
        toast.error('图片上传失败，请重试');
      } finally {
        setUploading(prev => ({ ...prev, [index]: false }));
      }
    }

    // 重置 input
    if (fileInputRef.current) {
      fileInputRef.current.value = '';
    }
  };

  const removeImage = (index: number) => {
    onImagesChange(images.filter((_, i) => i !== index));
  };

  return (
    <div className="flex flex-wrap gap-2">
      {images.map((img, index) => (
        <div key={index} className="relative group">
          {uploading[index] ? (
            <div className="h-20 w-20 flex items-center justify-center rounded-lg border border-amber-200">
              <Loader2 className="w-6 h-6 text-amber-400 animate-spin" />
            </div>
          ) : (
            <img
              src={img.publicUrl}
              alt={`上传的图片 ${index + 1}`}
              className="h-20 w-20 object-cover rounded-lg border border-amber-200"
            />
          )}
          {!disabled && !uploading[index] && (
            <button
              onClick={() => removeImage(index)}
              className="absolute -top-2 -right-2 bg-red-500 text-white rounded-full p-1 opacity-0 group-hover:opacity-100 transition-opacity"
            >
              <X size={14} />
            </button>
          )}
        </div>
      ))}

      {!disabled && images.length < maxImages && (
        <button
          onClick={() => fileInputRef.current?.click()}
          className="h-20 w-20 flex flex-col items-center justify-center rounded-lg border-2 border-dashed border-amber-300 hover:border-amber-400 transition-colors"
          disabled={Object.values(uploading).some(Boolean) || !messageId}
        >
          <ImageIcon className="w-6 h-6 text-amber-400" />
          <span className="text-xs text-amber-400 mt-1">上传</span>
        </button>
      )}

      <input
        ref={fileInputRef}
        type="file"
        accept="image/*"
        multiple
        className="hidden"
        onChange={handleFileSelect}
      />
    </div>
  );
};
```

#### 2.6 更新 chatStore

**文件**：`frontend/src/store/chatStore.ts`

```typescript
interface ChatStore {
  messages: ChatMessage[];
  currentSessionKey: string;
  isTyping: boolean;

  // 新增：当前消息的图片列表
  currentImages: ImageAttachment[];

  sendMessage: (content: string, images?: ImageAttachment[]) => Promise<void>;
  setCurrentImages: (images: ImageAttachment[]) => void;
  loadHistory: (sessionKey: string) => Promise<void>;
}

export const useChatStore = create<ChatStore>((set, get) => ({
  messages: [],
  currentSessionKey: '',
  isTyping: false,
  currentImages: [],

  /**
   * 发送消息（支持图片）
   */
  sendMessage: async (content: string, images?: ImageAttachment[]) => {
    const sessionKey = get().currentSessionKey;
    const messageImages = images || get().currentImages;

    // 1. 添加用户消息到本地状态
    const userMessage: ChatMessage = {
      role: 'user',
      content,
      timestamp: Date.now(),
      images: messageImages.length > 0 ? messageImages : undefined
    };
    set({ messages: [...get().messages, userMessage], isTyping: true });

    // 2. 发送 WebSocket 消息
    webSocketManager.send({
      type: 'CHAT',
      content,
      sessionId: sessionKey,
      images: messageImages.length > 0 ? messageImages : undefined
    });

    // 3. 清空当前图片列表
    set({ currentImages: [] });
  },

  /**
   * 设置当前消息的图片列表
   */
  setCurrentImages: (images: ImageAttachment[]) => {
    set({ currentImages: images });
  },

  /**
   * 加载会话历史（含图片）
   */
  loadHistory: async (sessionKey: string) => {
    try {
      const response = await chatApi.getSessionMessagesWithImages(sessionKey);
      const historyMessages = response.data.map(msg => ({
        ...msg,
        timestamp: new Date(msg.createdAt).getTime()
      }));

      set({
        messages: historyMessages,
        currentSessionKey: sessionKey
      });
    } catch (error) {
      console.error('加载历史消息失败', error);
      toast.error('加载历史消息失败');
    }
  }
}));
```

```typescript
import { ImageUploader } from './ImageUploader';

    e.preventDefault();

    if ((!input.trim() && pendingImages.length === 0) || disabled) {
```
