# 看板状态转换限制方案

## 1. 当前问题

前端看板使用 `dnd-kit` 实现拖拽，目前**任意状态可以拖拽到任意状态**，没有验证转换是否合法。

```tsx
// BoardView.tsx 当前实现 - 无验证
const handleDragEnd = async (event: DragEndEvent) => {
  const { active, over } = event;
  if (over && active.id !== over.id) {
    const applicationId = active.id as number;
    const newStatus = over.id as keyof typeof STATUS_CONFIG;
    // 直接调用，无验证！
    await updateApplicationStatus(applicationId, newStatus);
  }
};
```

## 2. 解决方案对比

### 方案 A：前端硬编码规则

**优点**：实现简单，无需后端改动
**缺点**：规则分散，维护困难

### 方案 B：后端提供规则 API（推荐）

**优点**：规则统一管理，前后端一致
**缺点**：需要新增 API

### 方案 C：混合方案

**优点**：前端缓存规则，减少请求
**缺点**：复杂度稍高

---

## 3. 推荐方案 B：详细设计

### 3.1 整体流程

```
┌─────────────────────────────────────────────────────────────────────┐
│                        拖拽流程                                      │
└─────────────────────────────────────────────────────────────────────┘

用户拖拽卡片
    │
    ▼
前端获取卡片当前状态
    │
    ▼
前端检查目标状态是否在合法列表中 ──► 否 ──► 阻止拖拽，显示提示
    │
    │ 是
    ▼
调用后端 API 更新状态
    │
    ▼
后端再次验证（双重保险）
    │
    ├──► 验证通过 ──► 更新成功
    │
    └──► 验证失败 ──► 返回错误，前端回滚
```

### 3.2 后端 API 设计

#### 获取合法状态转换

```java
// StatusTransitionController.java

@RestController
@RequestMapping("/api/data/status")
@RequiredArgsConstructor
public class StatusTransitionController {

    private final StatusTransitionService statusTransitionService;

    /**
     * 获取申请的下一个可能状态
     * GET /api/data/status/applications/{id}/next-statuses
     */
    @GetMapping("/applications/{id}/next-statuses")
    public ResponseEntity<ApiResponse<List<StatusOption>>> getNextStatuses(
            @PathVariable Long id) {
        JobApplication app = applicationMapper.selectById(id);
        if (app == null) {
            return ResponseEntity.notFound().build();
        }

        ApplicationStatus currentStatus = ApplicationStatus.fromCode(app.getStatus());
        List<ApplicationStatus> nextStatuses =
            statusTransitionService.getNextPossibleStatuses(currentStatus);

        List<StatusOption> options = nextStatuses.stream()
            .map(s -> new StatusOption(s.getCode(), s.getDescription()))
            .toList();

        return ResponseEntity.ok(ApiResponse.success(options));
    }

    /**
     * 获取所有状态的合法转换规则（前端可缓存）
     * GET /api/data/status/transitions
     */
    @GetMapping("/transitions")
    public ResponseEntity<ApiResponse<Map<String, List<String>>>> getAllTransitions() {
        Map<String, List<String>> transitions = new HashMap<>();

        for (ApplicationStatus status : ApplicationStatus.values()) {
            List<String> nextStatuses = statusTransitionService
                .getNextPossibleStatuses(status)
                .stream()
                .map(ApplicationStatus::getCode)
                .toList();
            transitions.put(status.getCode(), nextStatuses);
        }

        return ResponseEntity.ok(ApiResponse.success(transitions));
    }
}

// DTO
record StatusOption(String code, String label) {}
```

#### 更新状态 API（增强验证）

```java
// DataController.java 修改

@PutMapping("/applications/{id}/status")
public ResponseEntity<ApiResponse<Void>> updateApplicationStatus(
        @PathVariable Long id,
        @RequestParam String status) {

    JobApplication app = applicationMapper.selectById(id);
    if (app == null) {
        return ResponseEntity.notFound().build();
    }

    ApplicationStatus currentStatus = ApplicationStatus.fromCode(app.getStatus());
    ApplicationStatus newStatus = ApplicationStatus.fromCode(status);

    if (newStatus == null) {
        return ResponseEntity.badRequest()
            .body(ApiResponse.error("无效的状态: " + status));
    }

    // 验证状态转换是否合法
    if (!statusTransitionService.canTransition(currentStatus, newStatus)) {
        return ResponseEntity.badRequest()
            .body(ApiResponse.error(
                String.format("不允许从 %s 转换到 %s",
                    currentStatus.getDescription(),
                    newStatus.getDescription())));
    }

    // 执行更新
    statusTransitionService.transitionApplicationStatus(id, newStatus);

    return ResponseEntity.ok(ApiResponse.success(null));
}
```

### 3.3 前端实现

#### 3.3.1 状态转换规则 Hook

```typescript
// hooks/useStatusTransitions.ts
import { useQuery } from '@tanstack/react-query';
import { apiClient } from '@/services/apiClient';

// 状态转换规则类型
type TransitionRules = Record<string, string[]>;

// 获取所有状态转换规则
export function useStatusTransitions() {
  return useQuery({
    queryKey: ['status-transitions'],
    queryFn: async () => {
      const response = await apiClient.get('/status/transitions');
      return response.data.data as TransitionRules;
    },
    staleTime: 1000 * 60 * 30, // 缓存 30 分钟
  });
}

// 检查状态转换是否合法
export function canTransition(
  rules: TransitionRules | undefined,
  fromStatus: string,
  toStatus: string
): boolean {
  if (!rules) return true; // 规则未加载时允许（后端会验证）
  const allowedTargets = rules[fromStatus] || [];
  return allowedTargets.includes(toStatus);
}
```

#### 3.3.2 修改看板组件

```tsx
// BoardView.tsx 修改版

import { useStatusTransitions, canTransition } from '@/hooks/useStatusTransitions';
import { toast } from 'react-hot-toast';

export function BoardView({ isAiPanelOpen = true }: BoardViewProps) {
  const { applications, updateApplicationStatus, fetchApplications } = useApplicationStore();
  const { data: transitionRules } = useStatusTransitions();
  const [selectedApplication, setSelectedApplication] = useState<number | null>(null);
  const [dragOverStatus, setDragOverStatus] = useState<string | null>(null);
  const [draggingApp, setDraggingApp] = useState<JobApplication | null>(null);

  // ... 其他代码

  // 处理拖拽结束
  const handleDragEnd = async (event: DragEndEvent) => {
    const { active, over } = event;

    setDragOverStatus(null);
    setDraggingApp(null);

    if (!over || active.id === over.id) {
      return;
    }

    const applicationId = active.id as number;
    const newStatus = over.id as keyof typeof STATUS_CONFIG;
    const application = applications.find(app => app.id === applicationId);

    if (!application) return;

    // 前端验证状态转换是否合法
    if (!canTransition(transitionRules, application.status, newStatus)) {
      toast.error(`不允许从 "${STATUS_CONFIG[application.status as keyof typeof STATUS_CONFIG].label}"
                   转换到 "${STATUS_CONFIG[newStatus].label}"`);
      return;
    }

    try {
      await updateApplicationStatus(applicationId, newStatus);
      toast.success('状态更新成功');
    } catch (error: any) {
      // 后端验证失败，显示错误信息
      const errorMsg = error.response?.data?.message || '状态更新失败';
      toast.error(errorMsg);
      // 刷新列表回滚
      fetchApplications();
    }
  };

  // 拖拽开始
  const handleDragStart = (event: DragStartEvent) => {
    const applicationId = event.active.id as number;
    const application = applications.find(app => app.id === applicationId);
    setDraggingApp(application || null);
  };

  // 拖拽悬停
  const handleDragOver = (event: DragOverEvent) => {
    const { over } = event;
    setDragOverStatus(over?.id as string || null);
  };

  return (
    <div className="p-2 md:p-4 h-full overflow-hidden">
      <DndContext
        sensors={sensors}
        onDragStart={handleDragStart}
        onDragOver={handleDragOver}
        onDragEnd={handleDragEnd}
      >
        <div className="flex gap-1 md:gap-2 h-full overflow-x-auto">
          {(Object.keys(STATUS_CONFIG) as Array<keyof typeof STATUS_CONFIG>).map((status) => {
            // 判断当前列是否可以接收拖拽
            const canDrop = draggingApp
              ? canTransition(transitionRules, draggingApp.status, status)
              : true;

            return (
              <BoardColumn
                key={status}
                status={status}
                applications={groupedApplications[status]}
                isAiPanelOpen={isAiPanelOpen}
                onCardClick={handleCardClick}
                canDrop={canDrop}
                isDragOver={dragOverStatus === status && !canDrop}
              />
            );
          })}
        </div>
      </DndContext>
      {/* ... */}
    </div>
  );
}
```

─────────────────────────────────────────────────────────────────────────────

                 [通过筛选] [未通过] [撤回]

                 [进入终面] [收到Offer] [面试失败] [撤回]

FINAL_ROUND   →  OFFERED, REJECTED, WITHDRAWN
                 [收到Offer] [被拒绝] [撤回]

OFFERED       →  ACCEPTED, DECLINED, EXPIRED

  DECLINED    - 主动拒绝Offer
  EXPIRED     - Offer超时未处理
  REJECTED    - 被公司拒绝
  WITHDRAWN   - 主动撤回申请
─────────────────────────────────────────────────────────────────────────────

  FINAL_ROUND: ['OFFERED', 'REJECTED', 'WITHDRAWN'],

  // Offer 阶段
  OFFERED: ['ACCEPTED', 'DECLINED', 'EXPIRED'],

  // 终态：无法转换
  ACCEPTED: [],
  DECLINED: [],
  EXPIRED: [],
  REJECTED: [],
  WITHDRAWN: [],
};

/**
 * 检查状态转换是否合法
 */
export function isValidTransition(from: string, to: string): boolean {
  const allowed = STATUS_TRANSITIONS[from];
  return allowed ? allowed.includes(to) : false;
}

/**
 * 获取状态的下一个可能状态
 */
export function getNextStatuses(currentStatus: string): string[] {
  return STATUS_TRANSITIONS[currentStatus] || [];
}

/**
 * 判断状态是否为终态
 */
export function isTerminalStatus(status: string): boolean {
  const transitions = STATUS_TRANSITIONS[status];
  return transitions !== undefined && transitions.length === 0;
}

/**
 * 判断状态是否可以创建面试安排
 * 只有在"已进入流程"的状态才能安排面试
 */
export function canScheduleInterview(status: string): boolean {
  const allowedStatuses = [
    'SCREENING',     // 筛选中 - 可能随时被安排面试
    'INTERVIEW',     // 面试中 - 可以安排更多轮
    'FINAL_ROUND',   // 终面中 - 可以安排 HR 面等
    'OFFERED',       // 已收到Offer - 可能有后续沟通
  ];
  return allowedStatuses.includes(status);
}

/**
 * 获取不允许创建面试的原因
 */
export function getInterviewDisabledReason(status: string): string | null {
  if (canScheduleInterview(status)) return null;

  const reasons: Record<string, string> = {
    WISHLIST: '尚未投递申请，无法安排面试',
    APPLIED: '申请刚提交，请等待 HR 筛选后再安排面试',
    ACCEPTED: '已接受 Offer，无需再安排面试',
    DECLINED: '已拒绝 Offer，无法安排面试',
    EXPIRED: 'Offer 已过期，无法安排面试',
    REJECTED: '申请已被拒绝，无法安排面试',
    WITHDRAWN: '已撤回申请，无法安排面试',
  };

  return reasons[status] || '当前状态无法安排面试';
}
