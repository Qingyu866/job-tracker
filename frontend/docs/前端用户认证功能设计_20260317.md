## 一、设计概述

### 1.1 认证流程

```
┌─────────────┐      ┌─────────────┐      ┌──────────────┐
│   用户 UI    │      │  前端应用   │      │   后端 API     │
└──────┬──────┘      └──────┬──────┘      └──────┬───────┘
       │                    │                     │
       │  1. 访问需要认证的页面 │                     │
       │ ──────────────────>│                     │
       │                    │                     │
       │  2. 检查是否有 Token  │                     │
       │                    │                     │
       │  3. 没有 Token      │                     │
       │ ──────────────────────────────────────>│
       │                    │                     │
       │  4. 跳转到登录页    │                     │
       │ <─────────────────│                     │
       │                    │                     │
       │  5. 输入用户名密码   │                     │
       │ ──────────────────────────────────────>│
       │                    │                     │
       │                    │  6. POST /auth/login             │
       │                    │ ────────────────────>│
       │                    │                     │
       │                    │  7. 返回 Token + 用户信息        │
       │                    │ <────────────────────│
       │                    │                     │
       │  8. 保存 Token 到 localStorage                   │
       │ <─────────────────│                     │
       │                    │                     │
       │  9. 更新用户状态     │                     │
       │                    │ ───────────────────>│
       │                    │                     │
       │  10. 跳转回原页面     │                     │
       │  <─────────────────│                     │
       │                    │                     │
       │  11. 后续请求携带 Token                        │
       │ ──────────────────────────────────────>│
```

### 1.2 核心功能

| 功能 | 说明 |
|------|------|
| 登录 | 用户名密码登录，返回 Token |
| 注册 | 新用户注册，自动登录 |
| 退出 | 清除 Token，跳转登录页 |
| Token 刷新 | 自动续期（7天有效期） |
| 权限校验 | 401 自动跳转登录页 |

## 二、Token 管理

### 2.1 authUtil 工具类

```typescript
// src/utils/auth.ts

const TOKEN_KEY = 'satoken';
const USER_INFO_KEY = 'userinfo';

export const authUtil = {
  // ========== Token 管理 ==========

  /**
   * 获取 Token
   */
  getToken(): string | null {
    return localStorage.getItem(TOKEN_KEY);
  },

  /**
   * 设置 Token
   */
  setToken(token: string): void {
    localStorage.setItem(TOKEN_KEY, token);
  },

  /**
   * 移除 Token
   */
  removeToken(): void {
    localStorage.removeItem(TOKEN_KEY);
  },

  /**
   * 判断是否已登录
   */
  isLoggedIn(): boolean {
    return !!this.getToken();
  },

  // ========== 用户信息管理 ==========

  /**
   * 保存用户信息
   */
  saveUserInfo(userInfo: UserInfo): void {
    localStorage.setItem(USER_INFO_KEY, JSON.stringify(userInfo));
  },

  /**
   * 获取用户信息
   */
  getUserInfo(): UserInfo | null {
    const info = localStorage.getItem(USER_INFO_KEY);
    return info ? JSON.parse(info) : null;
  },

  /**
   * 清除用户信息
   */
  clearUserInfo(): void {
    localStorage.removeItem(USER_INFO_KEY);
  },

  // ========== 认证操作 ==========

  /**
   * 登录
   */
  async login(username: string, password: string): Promise<LoginResponse> {
    const response = await apiClient.post<LoginResponse>('/auth/login', {
      username,
      password
    });

    // 保存 Token
    this.setToken(response.data.token);

    // 保存用户信息
    this.saveUserInfo(response.data.userInfo);

    return response.data;
  },

  /**
   * 注册
   */
  async register(data: RegisterRequest): Promise<void> {
    await apiClient.post('/auth/register', data);
  },

  /**
   * 退出登录
   */
  async logout(): Promise<void> {
    try {
      await apiClient.post('/auth/logout');
    } finally {
      this.removeToken();
      this.clearUserInfo();
    }
  },

  /**
   * 修改密码
   */
  async changePassword(oldPassword: string, newPassword: string): Promise<void> {
    await apiClient.post('/auth/change-password', {
      oldPassword,
      newPassword
    });
  }
};
```

## 三、API 拦截器

### 3.1 请求拦截器 - 自动添加 Token

```typescript
// src/lib/apiClient.ts

import axios, { AxiosError, InternalAxiosRequestConfig } from 'axios';
import { authUtil } from '@/utils/auth';

const apiClient = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL || 'http://localhost:8080/api',
  timeout: 30000
});

// 请求拦截器：自动添加 Token
apiClient.interceptors.request.use(
  (config: InternalAxiosRequestConfig) => {
    const token = authUtil.getToken();

    // 如果有 Token，自动添加到请求头
    if (token && config.headers) {
      config.headers.Authorization = `Bearer ${token}`;
    }

    return config;
  },
  (error: AxiosError) => {
    return Promise.reject(error);
  }
);
```

### 3.2 响应拦截器 - 处理 401

```typescript
// 响应拦截器：处理未登录
apiClient.interceptors.response.use(
  (response) => response,
  (error: AxiosError) => {
    // 处理 401 未登录错误
    if (error.response?.status === 401) {
      // 清除过期的 Token
      authUtil.removeToken();
      authUtil.clearUserInfo();

      // 跳转到登录页
      const currentPath = window.location.pathname;
      if (currentPath !== '/login' && currentPath !== '/register') {
        // 保存当前路径，登录后跳回
        sessionStorage.setItem('redirect_after_login', currentPath);
        window.location.href = '/login';
      }
    }

    return Promise.reject(error);
  }
);
```

## 四、用户状态管理

### 4.1 userStore

```typescript
// src/store/userStore.ts

interface UserState {
  // 用户信息
  userInfo: UserInfo | null;

  // 登录状态
  isLoggedIn: boolean;

  // 登录
  login: (username: string, password: string) => Promise<void>;

  // 注册
  register: (data: RegisterRequest) => Promise<void>;

  // 退出登录
  logout: () => Promise<void>;

  // 获取用户信息
  fetchUserInfo: () => Promise<void>;

  // 修改密码
  changePassword: (oldPassword: string, newPassword: string) => Promise<void>;
}

interface UserInfo {
  id: number;
  username: string;
  nickname: string;
  avatar: string | null;
  email: string | null;
  phone: string | null;
}

export const useUserStore = create<UserState>((set) => ({
  userInfo: authUtil.getUserInfo(),
  isLoggedIn: authUtil.isLoggedIn(),

  login: async (username: string, password: string) => {
    try {
      const response = await authUtil.login(username, password);

      set({
        userInfo: response.userInfo,
        isLoggedIn: true
      });

      // 跳转回之前的页面或首页
      const redirectPath = sessionStorage.getItem('redirect_after_login');
      sessionStorage.removeItem('redirect_after_login');

      if (redirectPath && redirectPath !== '/login') {
        window.location.href = redirectPath;
      } else {
        window.location.href = '/';
      }
    } catch (error: any) {
      throw error;
    }
  },

  register: async (data: RegisterRequest) => {
    try {
      await authUtil.register(data);

      // 注册成功后自动登录
      toast.success('注册成功，正在登录...');
      await useUserStore.getState().login(data.username, data.password!);
    } catch (error: any) {
      throw error;
    }
  },

  logout: async () => {
    try {
      await authUtil.logout();

      set({
        userInfo: null,
        isLoggedIn: false
      });

      // 跳转到登录页
      window.location.href = '/login';
    } catch (error: any) {
      // 即使退出失败也要清除本地状态
      set({
        userInfo: null,
        isLoggedIn: false
      });
      window.location.href = '/login';
    }
  },

  fetchUserInfo: async () => {
    try {
      const response = await apiClient.get<{ data: UserInfo }>('/auth/info');

      set({
        userInfo: response.data.data,
        isLoggedIn: true
      });

      // 更新本地存储
      authUtil.saveUserInfo(response.data.data);
    } catch (error: any) {
      // 401 错误由拦截器处理
      throw error;
    }
  },

  changePassword: async (oldPassword: string, newPassword: string) => {
    try {
      await authUtil.changePassword(oldPassword, newPassword);

      toast.success('密码修改成功');
    } catch (error: any) {
      throw error;
    }
  }
}));
```

```tsx
// src/pages/LoginPage.tsx

import { useState } from 'react';
import { useNavigate } from 'react-router-dom';
import { useUserStore } from '@/store/userStore';

function LoginPage() {
  const navigate = useNavigate();
  const { login, isLoggedIn } = useUserStore();
  const [username, setUsername] = useState('');
  const [password, setPassword] = useState('');
  const [loading, setLoading] = useState(false);

  // 如果已登录，跳转到首页
  useEffect(() => {
    if (isLoggedIn) {
      navigate('/');
    }
  }, [isLoggedIn, navigate]);

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();

    if (!username || !password) {
      toast.error('请输入用户名和密码');
      return;
    }

    setLoading(true);
    try {
      await login(username, password);
      toast.success('登录成功');
    } catch (error: any) {
      toast.error('登录失败', {
        description: error.response?.data?.message || error.message
      });
    } finally {
      setLoading(false);
    }
  };

  return (
    <div className="login-page min-h-screen flex items-center justify-center bg-gray-50">
      <div className="login-card w-full max-w-md bg-white rounded-lg shadow-lg p-8">
        {/* Logo 或标题 */}
        <div className="text-center mb-8">
          <h1 className="text-2xl font-bold text-gray-900">Job Tracker</h1>
          <p className="text-gray-500 mt-2">欢迎回来</p>
        </div>

        {/* 登录表单 */}
        <form onSubmit={handleSubmit} className="space-y-4">
          <div>
            <label className="block text-sm font-medium text-gray-700 mb-1">
              用户名
            </label>
            <input
              type="text"
              value={username}
              onChange={(e) => setUsername(e.target.value)}
              className="w-full px-3 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500"
              placeholder="请输入用户名"
              autoComplete="username"
              required
            />
          </div>

            <label className="block text-sm font-medium text-gray-700 mb-1">
              密码

          </a>

      </div>

  );
```

### 5.2 RegisterPage 组件

```tsx
// src/pages/RegisterPage.tsx

function RegisterPage() {
  const navigate = useNavigate();
  const { register } = useUserStore();

  const [formData, setFormData] = useState({
    username: '',
    password: '',
    confirmPassword: '',
    nickname: '',
    email: ''
  });
  const [loading, setLoading] = useState(false);

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();

    // 验证
    if (formData.password.length < 6) {
      toast.error('密码长度不能少于6位');

      return;

      await register({

        password: formData.password,
        nickname: formData.nickname || formData.username,

      toast.error('注册失败', {

      setLoading(false);
    }

  return (
    <div className="register-page min-h-screen flex items-center justify-center bg-gray-50 py-12">
      <div className="register-card w-full max-w-md bg-white rounded-lg shadow-lg p-8">
```

### 6.1 路由守卫

```typescript
// src/App.tsx

import { useUserStore } from '@/store/userStore';

function App() {
  const { isLoggedIn, fetchUserInfo } = useUserStore();

  useEffect(() => {
    // 应用启动时检查登录状态
    const initAuth = async () => {
      if (authUtil.isLoggedIn()) {
        try {
          await fetchUserInfo();
        } catch {
          // Token 无效，清除本地状态
          authUtil.removeToken();
          authUtil.clearUserInfo();
        }
      }
    };

    initAuth();
  }, []);

  return (
    <Router>
      <Routes>
        {/* 公开路由 */}
        <Route path="/login" element={<LoginPage />} />
        <Route path="/register" element={<RegisterPage />} />

        {/* 需要认证的路由 */}
        <Route
          path="/*"
          element={
            <ProtectedRoute>
              <WorkspacePage />
            </ProtectedRoute>
          }
        />
      </Routes>
    </Router>
  );
}

// 受保护的路由组件
function ProtectedRoute({ children }: { children: React.ReactNode }) {
  const { isLoggedIn } = useUserStore();

  if (!isLoggedIn) {
    return <Navigate to="/login" replace />;
  }

  return <>{children}</>;
}
```

## 九、API 接口

### 9.1 认证接口

```typescript
// src/services/authApi.ts

export interface LoginRequest {
  username: string;
  password: string;
}

export interface LoginResponse {
  token: string;
  tokenName: string;
  userInfo: UserInfo;
}

export interface RegisterRequest {
  username: string;
  password: string;
  nickname?: string;
  email?: string;
  phone?: string;
}

export interface UserInfo {
  id: number;
  username: string;
  nickname: string;
  avatar: string | null;
  email: string | null;
  phone: string | null;
}

export const authApi = {
  /**
   * 用户登录
   */
  async login(data: LoginRequest): Promise<AxiosResponse<ApiResponse<LoginResponse>>> {
    return apiClient.post('/auth/login', data);
  },

  /**
   * 用户注册
   */
  async register(data: RegisterRequest): Promise<AxiosResponse<ApiResponse<void>>> {
    return apiClient.post('/auth/register', data);
  },

  /**
   * 获取当前用户信息
   */
  async getCurrentUserInfo(): Promise<AxiosResponse<ApiResponse<UserInfo>>> {
    return apiClient.get('/auth/info');
  },

  /**
   * 退出登录
   */
  async logout(): Promise<AxiosResponse<ApiResponse<void>>> {
    return apiClient.post('/auth/logout');
  },

  /**
   * 修改密码
   */
  async changePassword(data: {
    oldPassword: string;
    newPassword: string;
  }): Promise<AxiosResponse<ApiResponse<void>>> {
    return apiClient.post('/auth/change-password', data);
  }
};
```

## 十一、常见问题

### Q1: Token 过期了怎么办？

**A**:
- API 响应 401 时，自动清除过期 Token
- 自动跳转到登录页
- 登录后自动跳转回原页面

### Q2: 如何实现"记住我"功能？

**A**:
```typescript
// 登录时选择"记住我"
const login = async (username: string, password: string, remember: boolean) => {
  const response = await authApi.login({ username, password });

  // 保存 Token
  if (remember) {
    localStorage.setItem('satoken', response.data.token);
  } else {
    sessionStorage.setItem('satoken', response.data.token);
  }
};
```

### Q3: 如何在多个标签页同步登录状态？

**A**:
- 使用 `BroadcastChannel` API
- 登录/退出时广播事件
- 其他标签页接收事件并更新状态
