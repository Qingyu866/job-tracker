
### 3.1 ImageDropZone 组件

**新增文件**: `src/components/ocr/ImageDropZone.tsx`

```tsx
interface ImageDropZoneProps {
  accept: string;
  onUpload: (file: File) => void;
  ocrMode?: 'resume' | 'jd' | 'general';
  maxSize?: number;
}

function ImageDropZone({
  accept,
  onUpload,
  ocrMode = 'general',
  maxSize = 10 * 1024 * 1024
}: ImageDropZoneProps) {
  const [dragging, setDragging] = useState(false);
  const [processing, setProcessing] = useState(false);

  const handleDrop = async (e: React.DragEvent) => {
    e.preventDefault();
    setDragging(false);

    const file = e.dataTransfer.files[0];
    if (!validateFile(file)) return;

    setProcessing(true);
    try {
      await onUpload(file);
    } finally {
      setProcessing(false);
    }
  };

  const validateFile = (file: File): boolean => {
    // 验证文件类型
    if (!accept.split(',').some(type => file.type.match(type))) {
      toast.error('不支持的文件类型');
      return false;
    }

    // 验证文件大小
    if (file.size > maxSize) {
      toast.error(`文件大小不能超过 ${maxSize / 1024 / 1024}MB`);
      return false;
    }

    return true;
  };

  return (
    <div
      className={cn('image-drop-zone', dragging && 'dragging', processing && 'processing')}
      onDragOver={(e) => { e.preventDefault(); setDragging(true); }}
      onDragLeave={() => setDragging(false)}
      onDrop={handleDrop}
    >
      <input
        type="file"
        accept={accept}
        onChange={(e) => {
          const file = e.target.files?.[0];
          if (file) handleDrop({ dataTransfer: { files: [file] } } as any);
        }}
        className="hidden-input"
      />

      {processing ? (
        <div className="processing-content">
          <Spinner />
          <p>OCR 识别中...</p>
        </div>
      ) : (
        <div className="drop-content">
          <Upload className="icon" />
          <p>拖拽图片到此处</p>
          <p className="hint">或点击选择文件</p>
          <p className="formats">支持: JPG, PNG, PDF</p>
        </div>
      )}
    </div>
  );
}
```

```tsx
interface OcrResultPreviewProps {
  result: OcrResult;
  mode: 'resume' | 'jd' | 'general';
  onConfirm?: (data: any) => void;
}

function OcrResultPreview({ result, mode, onConfirm }: OcrResultPreviewProps) {
  if (result.status === 'pending') {
    return (
      <div className="ocr-preview ocr-preview--pending">
        <Spinner />
        <p>正在识别图片内容...</p>
      </div>
    );
  }

  if (result.status === 'failed') {
    return (
      <div className="ocr-preview ocr-preview--failed">
        <AlertCircle className="icon" />
        <p>识别失败</p>
        <p className="error">{result.error}</p>
        <Button variant="outline" size="sm">重新识别</Button>
      </div>
    );
  }

  return (
    <div className="ocr-preview ocr-preview--success">
      <div className="preview-header">
        <CheckCircle className="icon success" />
        <span>识别成功</span>
        <Badge variant="secondary">置信度: {(result.confidence! * 100).toFixed(0)}%</Badge>
      </div>

      {mode === 'resume' && (
        <ResumeDataPreview data={result.resumeData} />
      )}

      {mode === 'jd' && (
        <JdDataPreview data={result.jdData} />
      )}

      {mode === 'general' && (
        <TextPreview text={result.text} />
      )}

      {onConfirm && (
        <div className="preview-actions">
          <Button variant="outline" onClick={() => window.location.reload()}>
            重新上传
          </Button>
          <Button onClick={() => onConfirm(result.data)}>
            确认使用
          </Button>
        </div>
      )}
    </div>
  );
}
```

## 四、API 集成

### 4.1 OCR API 服务

**新增文件**: `src/services/ocrApi.ts`

```typescript
export const ocrApi = {
  /**
   * 识别图片中的文字
   */
  async recognizeImage(imageData: string): Promise<OcrResult> {
    return request.post('/ocr/recognize', { imageData });
  },

  /**
   * 识别简历并解析结构化信息
   */
  async recognizeResume(file: File): Promise<ResumeInfo> {
    const formData = new FormData();
    formData.append('file', file);
    return request.post('/ocr/resume', formData, {
      headers: { 'Content-Type': 'multipart/form-data' }
    });
  },

  /**
   * 识别 JD 并提取技能
   */
  async recognizeJd(file: File, applicationId?: number): Promise<JdInfo> {
    const formData = new FormData();
    formData.append('file', file);
    if (applicationId) formData.append('applicationId', applicationId);
    return request.post('/ocr/jd', formData, {
      headers: { 'Content-Type': 'multipart/form-data' }
    });
  },

  /**
   * 获取 OCR 调用记录
   */
  async getRecords(page = 0, size = 20): Promise<OcrRecord[]> {
    return request.get('/ocr/records', { params: { page, size } });
  }
};
```

### 5.1 OCR Store

**新增文件**: `src/store/ocrStore.ts`

```typescript
interface OcrState {
  // 当前上传的图片
  currentImage: {
    dataUrl: string;
    name: string;
    type: 'resume' | 'jd' | 'general';
  } | null;

  // OCR 识别结果
  ocrResult: {
    status: 'idle' | 'pending' | 'success' | 'failed';
    text?: string;
    confidence?: number;
    resumeData?: ResumeInfo;
    jdData?: JdInfo;
    error?: string;
  } | null;

  // 识别历史
  records: OcrRecord[];
}

interface OcrStore extends OcrState {
  // 操作
  setCurrentImage: (image: OcrState['currentImage']) => void;
  clearCurrentImage: () => void;
  recognizeImage: (imageData: string, type: 'resume' | 'jd' | 'general') => Promise<void>;
  clearResult: () => void;
  fetchRecords: () => Promise<void>;
}

export const useOcrStore = create<OcrStore>((set) => ({
  currentImage: null,
  ocrResult: null,
  records: [],

  setCurrentImage: (image) => set({ currentImage: image }),

  clearCurrentImage: () => set({ currentImage: null, ocrResult: null }),

  recognizeImage: async (imageData: string, type) => {
    set({ ocrResult: { status: 'pending' } });

    try {
      const result = await ocrApi.recognizeImage(imageData);
      set({ ocrResult: result });
    } catch (error) {
      set({ ocrResult: { status: 'failed', error: String(error) });
    }
  },

  clearResult: () => set({ ocrResult: null }),

  fetchRecords: async () => {
    const records = await ocrApi.getRecords();
    set({ records });
  }
}));
```

### 7.2 图片验证

```typescript
function validateImageFile(file: File): { valid: boolean; error?: string } {
  // 1. 检查文件大小
  const MAX_SIZE = 10 * 1024 * 1024; // 10MB
  if (file.size > MAX_SIZE) {
    return { valid: false, error: '图片大小不能超过 10MB' };
  }

  // 2. 检查文件类型
  const ACCEPTED_TYPES = ['image/jpeg', 'image/png', 'image/jpg', 'application/pdf'];
  if (!ACCEPTED_TYPES.includes(file.type)) {
    return { valid: false, error: '不支持的文件类型，请上传 JPG、PNG 或 PDF' };
  }

  // 3. 检查图片尺寸（可选）
  return new Promise<{ valid: boolean; error?: string }>((resolve) => {
    const img = new Image();
    img.onload = () => {
      if (img.width > 4096 || img.height > 4096) {
        resolve({ valid: false, error: '图片尺寸过大，建议使用 4096x4096 以下的图片' });
      } else {
        resolve({ valid: true });
    };
    img.onerror = () => resolve({ valid: false, error: '图片文件损坏' });
    img.src = URL.createObjectURL(file);
  });
}
```

### 9.1 图片上传流程

```
用户选择图片
    ↓
前端验证（大小、类型、尺寸）
    ↓

    ↓

AI 根据识别结果回复
```
